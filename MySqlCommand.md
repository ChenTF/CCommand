### 登录数据库
mysql -h (主机地址) -u (用户名) －p (用户密码)
### 查看数据库
show databases;
### 创建数据库
create database xbd(数据库名);
### 删除数据库
drop database xbd(数据库名);
### 使用数据库
use database xbd(数据库名);
### 创建表
命令：create table <表名> ( <字段名1> <类型1> [,..<字段名n> <类型n>]);
### 删除表
drop table person(表名);
